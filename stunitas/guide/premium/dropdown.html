<script type="text/javascript">
    $(function(){
        $('.prem__dropdown--btn1').cnDropdown({
            width: 'auto',
            align: 'left',
            onMenuClick: function(menuEl, menuIndex, btnEl){ // 메뉴 엘리먼트, 메뉴 인덱스, 버튼 엘리먼트
                const menuHTML = menuEl.find('a').html().trim();
                btnEl.find('.select').html(menuHTML);
                btnEl.find('input[type=hidden]').val(menuEl.data('value'));
            }
        })

        $('.prem__dropdown--btn2').cnDropdown({
            width: 'auto',
            align: 'left',
            onMenuClick: function(menuEl, menuIndex, btnEl){ // 메뉴 엘리먼트, 메뉴 인덱스, 버튼 엘리먼트
                const menuHTML = menuEl.find('a').html().trim();
                btnEl.addClass('selected').find('.select').html(menuHTML);
                btnEl.find('input[type=hidden]').val(menuEl.data('value'));
            }
        })
    })
</script>
