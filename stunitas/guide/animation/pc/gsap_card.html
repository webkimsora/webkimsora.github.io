<script type="text/javascript">
    $(function(){
        var controller = new ScrollMagic.Controller();

        var wipeAnimation = gsap.timeline()
            .add("first")
            .to(".low01", { x: -360, rotation: 0 }, "first")
            .to(".low02", { x: -135, rotation: 0 }, "first")
            .to(".low03", { x: 130, rotation: 0 }, "first")
            .to(".low04", { x: 360, rotation: 0 }, "first");

        var scene2 = new ScrollMagic.Scene({
            triggerElement: ".cont_low",
            triggerHook: 0,
            duration: "100%"
        })
        .setPin('.cont_low')
        .setTween(wipeAnimation)
        .addTo(controller);
    })
</script>
