<script type="text/javascript">
    $(function(){
        // ScrollMagic Controller 생성
        var controller = new ScrollMagic.Controller();

        // 각 .cont_easy 섹션에 대해 ScrollMagic Scene 설정
        var wipeAnimation = gsap.timeline()
            .to(".easy_img1" , 1, { x: '90%', y: '90%', rotation: 40, opacity: 0 })
            .to(".easy_img2" , 1, { x: '120%', y: '200%', rotation: 20, opacity: 0 })
            .to(".easy_img3" , 1, { x: '90%', y: '220%', rotation: 20, opacity: 0 })
            .to(".easy_img4" , 1, { x: '-170%', y: '90%', rotation: -60, opacity: 0 })
            .to(".easy_img5" , 1, { x: '-150%', y: '90%', rotation: -40, opacity: 0 })

        new ScrollMagic.Scene({
            triggerElement: ".cont_easy",
            triggerHook: 0, 
            duration: "300%"
        })
        
        .setPin(".cont_easy")
        .setTween(wipeAnimation) 
        .addTo(controller);
    });
</script>
